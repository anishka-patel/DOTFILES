** Yay
 * Get ~base-devel~ and ~git~
#+begin_src bash
sudo pacman -S --needed base-devel git --noconfirm
#+end_src
 * Get ~yay~ repository
#+begin_src bash
git clone https://aur.archlinux.org/yay.git
cd yay
#+end_src
 * Install ~yay~ with ~makepkg~
#+begin_src bash
makepkg -si
#+end_src
 * Check version
#+begin_src bash
yay --version
#+end_src
